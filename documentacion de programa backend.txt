🧠 1. CONCEPTO MULTITENANT APLICADO
👉 Cada registro pertenece a una tienda (tenant)
tenant_id UUID NOT NULL

📌 Esto permite:
Aislamiento total de datos
Escalabilidad
Seguridad con RLS (Row Level Security)

🏪 2. NUEVA TABLA PRINCIPAL: TENANTS
CREATE TABLE tenants (
  tenant_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  slug VARCHAR(100) UNIQUE NOT NULL,
  owner_id UUID,
  created_at TIMESTAMPTZ DEFAULT NOW()
);


👤 3. USUARIOS (MULTITENANT)
CREATE TABLE users (
  user_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(tenant_id) ON DELETE CASCADE,
  email VARCHAR(255) UNIQUE NOT NULL,
  password TEXT NOT NULL,
  role VARCHAR(50) DEFAULT 'customer',
  created_at TIMESTAMPTZ DEFAULT NOW()
);


🛍️ 4. PRODUCTOS
CREATE TABLE products (
  product_id SERIAL PRIMARY KEY,
  tenant_id UUID REFERENCES tenants(tenant_id) ON DELETE CASCADE,
  category_id INT,
  name VARCHAR(255),
  slug VARCHAR(280),
  description TEXT,
  base_price NUMERIC(12,2),
  is_active BOOLEAN DEFAULT TRUE,
  min_stock_alert INT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ
);


🎯 5. VARIANTES
CREATE TABLE product_variants (
  variant_id SERIAL PRIMARY KEY,
  tenant_id UUID REFERENCES tenants(tenant_id) ON DELETE CASCADE,
  product_id INT REFERENCES products(product_id) ON DELETE CASCADE,
  size_id INT,
  color_id INT,
  sku VARCHAR(100),
  price_override NUMERIC(12,2),
  stock INT,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ
);


🖼️ 6. IMÁGENES
CREATE TABLE product_images (
  image_id SERIAL PRIMARY KEY,
  tenant_id UUID REFERENCES tenants(tenant_id) ON DELETE CASCADE,
  product_id INT REFERENCES products(product_id) ON DELETE CASCADE,
  image_url TEXT,
  alt_text VARCHAR(255),
  display_order SMALLINT,
  is_primary BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);


🗂️ 7. CATEGORÍAS
CREATE TABLE categories (
  category_id SERIAL PRIMARY KEY,

  name VARCHAR(100),
  slug VARCHAR(120),
  description TEXT,
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ
);

🛒 9. CARRITO
CREATE TABLE carts (
  cart_id SERIAL PRIMARY KEY,
  tenant_id UUID REFERENCES tenants(tenant_id) ON DELETE CASCADE,
  session_token VARCHAR(128),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ
);

CREATE TABLE cart_items (
  cart_item_id SERIAL PRIMARY KEY,
  tenant_id UUID REFERENCES tenants(tenant_id) ON DELETE CASCADE,
  cart_id INT REFERENCES carts(cart_id) ON DELETE CASCADE,
  variant_id INT REFERENCES product_variants(variant_id),
  quantity INT,
  added_at TIMESTAMPTZ DEFAULT NOW()
);


📦 10. PEDIDOS
CREATE TABLE orders (
  order_id SERIAL PRIMARY KEY,

  order_number VARCHAR(20) UNIQUE,
  payment_method_id INT,
  shipping_name VARCHAR(200),
  shipping_address TEXT,
  shipping_city VARCHAR(100),
  shipping_phone VARCHAR(30),
  current_status VARCHAR(30),
  tracking_number VARCHAR(100),
  customer_email VARCHAR(255),

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ

🔐 15. SEGURIDAD (CLAVE EN SUPABASE)
RLS (Row Level Security)
Ejemplo:
CREATE POLICY "tenant isolation"
ON products
FOR ALL
USING (tenant_id = auth.jwt() ->> 'tenant_id');

👉 Esto asegura:
Un usuario SOLO ve su tienda
No hay filtraciones
